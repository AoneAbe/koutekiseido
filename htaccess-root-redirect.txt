# ============================================
# ルートから/appへのリダイレクト設定
# WordPressの.htaccessの先頭（# BEGIN WordPress の前）に追記してください
# ============================================

# ルートアクセスを/appにリダイレクト
<IfModule mod_rewrite.c>
  RewriteEngine On

  # ルート（/）へのアクセスを/appにリダイレクト
  RewriteRule ^$ /app/ [R=301,L]
</IfModule>
